Learn Solidity 4 Unit

December 11, 2022
  • solidity中的单位
uint表示uint256，表示0-2^256-1
// 科学计算器地址: https://www.desmos.com/scientific?lang=zh-CN
uint32 表示0-2^32-1
uint16
  • 转换例子
// SPDX-License-Identifier: MIT
pragma solidity >= 0.7.0 < 0.9.0;

// 从高到低
uint32 a = 0x12345678;
uint16 b = uint16(a); // b = 0x5678
// 从低到高
uint16 c = 0x1234;
uint32 d = uint32(c); // d = 0x00001234
// 同理 bytes2 和 bytes1
// 从高到低失去精度
bytes2 e = 0x1234;
bytes1 f = bytes1(e) // f = 0x12
// 从低到高补0
bytes2 g = 0x1234;
bytes4 h = bytes4(g); // h = 0x12340000
  • 以太坊单位换算
// SPDX-License-Identifier: MIT
pragma solidity >= 0.7.0 < 0.9.0;

contract learnEtherUnits {
    function test() public pure{
        // 10^18 wei == 1个以太
        assert(1000000000000000000 wei == 1 ether);
        assert(1 wei == 1);
        assert(1 ether == 1e18);
    }
}
  • 练习，在Solidity中声明时间和日期后缀
// SPDX-License-Identifier: MIT
pragma solidity >= 0.7.0 < 0.9.0;

contract learnEtherUnits {
    function exercise() public pure{
        assert(1 minutes == 60 seconds);
        assert(24 hours == 1440 minutes);
        assert(1 days == 24 hours);
        assert(1 weeks == 7 days);
    }
}
